    3.

    \[m^* = \frac{\hbar^2}{2a^2}\frac{1}{t\cos(ka)+4t'\cos(2ka)}\]

    Plot for t=t':

    k1 = np.linspace(-pi, -pi/2-0.01, 300);

pyplot.plot(k1, 1/(5*np.cos(k1)),'b');
pyplot.plot(k2, 1/(5*np.cos(k2)),'b');

pyplot.xlabel(r'$k$'); pyplot.ylabel('$m_{eff}(k)$');

pyplot.tight_layout();
